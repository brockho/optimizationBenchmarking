options(digits=22);

rcommPrintMatrix <- function(mat) {
  op   <- options(digits=22);
  
  dims <- dim(mat);
  m    <- dims[1];
  n    <- dims[2];
  
  cat(m);
  cat(' ');
  cat(n);
  cat('\n');
  
  for(i in 1:m) { 
    first <- TRUE;
    for(j in 1:n) {
      if (first) {
        first=FALSE;
      } else {
        cat(' ');
      }
      cat(mat[i,j]);
    }
    cat('\n');
  }
  options(op);
}
